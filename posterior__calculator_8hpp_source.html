124 if (is_conjugate_ && conjugate_posterior_) {

141 /**
142 * @brief Get credible interval
143 * @param confidence Confidence level (e.g., 0.95)
144 * @return Pair of (lower, upper) bounds
145 */
146 [[nodiscard]] std::pair<T, T> credible_interval(T confidence = 0.95) const {
147 if (is_conjugate_ && conjugate_posterior_) {
148 T alpha = (1 - confidence) / 2;
149
150 // Use inverse CDF for conjugate cases
151 auto [lower_bound, upper_bound] = conjugate_posterior_->support();
152
153 // Binary search for quantiles
154 auto find_quantile = [this](T target_cdf) {
155 auto [lower, upper] = conjugate_posterior_->support();
156
157 while (upper - lower > 1e-9) {
158 T mid = (lower + upper) / 2;
159 if (posterior_cdf(mid) < target_cdf) {
160 lower = mid;
161 } else {
162 upper = mid;
163 }
164 }
165 return (lower + upper) / 2;
166 };
167
168 return {find_quantile(alpha), find_quantile(1 - alpha)};
169 }
170
171 // Numerical approach for non-conjugate
172 return numerical_credible_interval(confidence);
173 }

175 /**
176 * @brief Get highest density interval (HDI)
177 * @param confidence Confidence level
178 * @return Pair of (lower, upper) bounds for HDI
179 */
180 [[nodiscard]] std::pair<T, T> highest_density_interval(T confidence = 0.95) const {
181 // HDI is the shortest interval containing the specified probability mass
182 // This is more complex than credible interval and requires optimization
183
184 auto [lower_support, upper_support] = get_integration_bounds();
185
186 // Define objective: minimize interval width subject to probability constraint
187 auto interval_width = [this, confidence, lower_support, upper_support](T lower) {
188 // Find upper bound such that P(lower < theta < upper) = confidence
189 T target_prob = confidence;
190 T current_prob = 0;
191 T upper = lower;
192 T step = (upper_support - lower_support) / 1000;
193
194 while (current_prob < target_prob && upper < upper_support) {
195 upper += step;
196 current_prob = posterior_cdf(upper) - posterior_cdf(lower);
197 }
198
199 return upper - lower;
200 };
201
202 // Find the lower bound that minimizes interval width
203 int bits = std::numeric_limits<T>::digits;
204 boost::uintmax_t max_iter = 100;
205
206 auto result = boost::math::tools::brent_find_minima(
207 interval_width, lower_support,
208 upper_support - (upper_support - lower_support) * confidence, bits, max_iter);
209
210 T optimal_lower = result.first;
211
212 // Find corresponding upper bound
213 T upper = optimal_lower;
214 T current_prob = 0;
215 T step = (upper_support - lower_support) / 1000;
216 while (current_prob < confidence && upper < upper_support) {
217 upper += step;
218 current_prob = posterior_cdf(upper) - posterior_cdf(optimal_lower);
219 }
220
221 return {optimal_lower, upper};
222 }

324 /**
325 * @brief Update using conjugate prior relationships
326 */
327 void update_conjugate(const vector_type& sufficient_stats) {
328 auto likelihood_family = likelihood_->family();
329
330 if (likelihood_family == "bernoulli" || likelihood_family == "binomial") {
331 // Beta-Binomial conjugacy
332 auto beta_prior = dynamic_cast<BetaPrior<T>*>(prior_.get());
333 if (beta_prior) {
334 T alpha = beta_prior->alpha();
335 T beta = beta_prior->beta();
336
337 // Update: alpha' = alpha + successes, beta' = beta + failures
338 T successes = sufficient_stats[0];
339 T trials = sufficient_stats[1];
340 T failures = trials - successes;
341
342 conjugate_posterior_ = std::make_unique<BetaPrior<T>>(alpha + successes, beta + failures);
343
344 // Update marginal likelihood
345 update_marginal_likelihood_conjugate(alpha, beta, successes, failures);
346 }
347 } else if (likelihood_family == "normal_known_variance") {
348 // Normal-Normal conjugacy
349 auto normal_prior = dynamic_cast<NormalPrior<T>*>(prior_.get());
350 auto normal_likelihood = dynamic_cast<NormalKnownVarianceLikelihood<T>*>(likelihood_.get());
351
352 if (normal_prior && normal_likelihood) {
353 T prior_mean = normal_prior->mean();
354 T prior_var = normal_prior->variance();
355 T likelihood_var = normal_likelihood->variance();
356
357 T sum_x = sufficient_stats[0];
358 T n = sufficient_stats[1];
359
360 // Update posterior parameters
361 T posterior_var = 1 / (1 / prior_var + n / likelihood_var);
362 T posterior_mean = posterior_var * (prior_mean / prior_var + sum_x / likelihood_var);
363
364 conjugate_posterior_
365 = std::make_unique<NormalPrior<T>>(posterior_mean, std::sqrt(posterior_var));
366 }
367 } else if (likelihood_family == "poisson") {
368 // Gamma-Poisson conjugacy
369 auto gamma_prior = dynamic_cast<GammaPrior<T>*>(prior_.get());
370 if (gamma_prior) {
371 T shape = gamma_prior->alpha();
372 T rate = gamma_prior->beta();
373
374 T sum_x = sufficient_stats[0];
375 T n = sufficient_stats[1];
376
377 // Update: shape' = shape + sum(x), rate' = rate + n
378 conjugate_posterior_ = std::make_unique<GammaPrior<T>>(shape + sum_x, rate + n);
379 }
380 }

383 /**
384 * @brief Update using numerical methods
385 */
386 void update_numerical(const vector_type& data) {
387 // Use importance sampling or MCMC
388 // This is a simplified version; real implementation would be more sophisticated
389
390 const std::size_t n_particles = 1000;
391 boost::random::mt19937 gen;
392 if (seed_) {
393 gen.seed(*seed_);
394 } else {
395 std::random_device rd;
396 gen.seed(rd());
397 }
398
399 // Generate particles from prior
400 auto prior_samples = prior_->sample(n_particles, gen);
401
402 // Calculate importance weights
403 importance_weights_.resize(n_particles);
404 T max_log_weight = -std::numeric_limits<T>::infinity();
405
406 for (std::size_t i = 0; i < n_particles; ++i) {
407 T theta = prior_samples[i];
408 T log_weight = likelihood_->log_evaluate(theta, data);
409 importance_weights_[i] = log_weight;
410 max_log_weight = std::max(max_log_weight, log_weight);
411 }
412
413 // Normalize weights (in log space for stability)
414 T log_sum_weights = 0;
415 for (auto& log_w : importance_weights_) {
416 log_w -= max_log_weight;
417 log_sum_weights = log_sum_exp(log_sum_weights, log_w);
418 }
419
420 // Convert to normalized weights
421 for (auto& log_w : importance_weights_) {
422 log_w = std::exp(log_w - log_sum_weights);
423 }
424
425 // Store samples
426 posterior_samples_ = prior_samples;
427
428 // Update marginal likelihood
429 log_marginal_likelihood_ += max_log_weight + log_sum_weights;
430 }

432 /**
433 * @brief Update marginal likelihood for conjugate cases
434 */
435 void update_marginal_likelihood_conjugate(T alpha, T beta, T successes, T failures) {
436 // For Beta-Binomial: use log Beta function
437 T log_ml = std::lgamma(alpha + beta) - std::lgamma(alpha) - std::lgamma(beta);
438 log_ml += std::lgamma(alpha + successes) + std::lgamma(beta + failures);
439 log_ml -= std::lgamma(alpha + beta + successes + failures);
440
441 log_marginal_likelihood_ += log_ml;
442 }

511 /**
512 * @brief Rejection sampling for posterior
513 */
514 [[nodiscard]] vector_type rejection_sample(std::size_t n_samples,
515 boost::random::mt19937& gen) const {
516 vector_type samples(n_samples);
517
518 // Find mode for envelope
519 T mode_theta = mode();
520 T mode_density = posterior_pdf(mode_theta);
521
522 auto [lower, upper] = get_integration_bounds();
523 boost::random::uniform_real_distribution<T> theta_dist(lower, upper);
524 boost::random::uniform_real_distribution<T> u_dist(0, 1);
525
526 std::size_t accepted = 0;
527 while (accepted < n_samples) {
528 T theta = theta_dist(gen);
529 T u = u_dist(gen);
530
531 T acceptance_prob = posterior_pdf(theta) / (mode_density * 1.1); // 10% safety margin
532
533 if (u < acceptance_prob) {
534 samples[accepted++] = theta;
535 }
536 }
537
538 return samples;
539 }

541 /**
542 * @brief Estimate effective sample size
543 */
544 [[nodiscard]] T estimate_ess() const {
545 if (importance_weights_.empty()) {
546 return static_cast<T>(n_observations_);
547 }
548
549 T sum_weights = 0;
550 T sum_weights_sq = 0;
551
552 for (T w : importance_weights_) {
553 sum_weights += w;
554 sum_weights_sq += w * w;
555 }
556
557 return (sum_weights * sum_weights) / sum_weights_sq;
558 }
