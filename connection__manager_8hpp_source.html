91 /**
92 * @brief Execute a transaction with automatic rollback on failure and timeout
93 * @param func Function to execute within transaction
94 * @param timeout_ms Maximum time in milliseconds (default: 30 seconds)
95 * @return Error code if transaction failed
96 */
97 template <typename Func>
98 boost::system::error_code execute_transaction(Func func, std::chrono::milliseconds timeout_ms
99 = std::chrono::milliseconds(30000)) {
100 auto conn = acquire_connection();
101 if (!conn) {
102 return boost::system::error_code(SQLITE_CANTOPEN, boost::system::generic_category());
103 }
104
105 // Set busy timeout on the connection
106 sqlite3_busy_timeout(conn->get(), static_cast<int>(timeout_ms.count()));
107
108 // Begin transaction with timeout tracking
109 auto start_time = std::chrono::steady_clock::now();
110 auto error = conn->begin_transaction();
111 if (error) {
112 return error;
113 }
114
115 try {
116 // Check if we've exceeded timeout
117 auto elapsed = std::chrono::steady_clock::now() - start_time;
118 if (elapsed > timeout_ms) {
119 conn->rollback();
120 BOOST_LOG_TRIVIAL(warning)
121 << "Transaction timed out after "
122 << std::chrono::duration_cast<std::chrono::milliseconds>(elapsed).count() << "ms";
123 return boost::system::error_code(SQLITE_BUSY, boost::system::generic_category());
124 }
125
126 // Execute user function
127 error = func(*conn);
128
129 if (error) {
130 conn->rollback();
131 return error;
132 }
133
134 // Check timeout again before commit
135 elapsed = std::chrono::steady_clock::now() - start_time;
136 if (elapsed > timeout_ms) {
137 conn->rollback();
138 BOOST_LOG_TRIVIAL(warning)
139 << "Transaction timed out during commit after "
140 << std::chrono::duration_cast<std::chrono::milliseconds>(elapsed).count() << "ms";
141 return boost::system::error_code(SQLITE_BUSY, boost::system::generic_category());
142 }
143
144 // Commit transaction
145 return conn->commit();
146
147 } catch (...) {
148 conn->rollback();
149 throw;
150 }
151 }

258 void start_io_context() {
259 io_context_ = std::make_unique<boost::asio::io_context>();
260 work_guard_ = std::make_unique<
261 boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
262 boost::asio::make_work_guard(*io_context_));
263
264 // Start worker threads
265 for (size_t i = 0; i < thread_pool_size_; ++i) {
266 worker_threads_.emplace_back([this] { io_context_->run(); });
267 }
268 }
269
270 /**
271 * @brief Stop IO context and join worker threads
272 */
273 void stop_io_context() {
274 if (work_guard_) {
275 work_guard_.reset();
276 }
277
278 if (io_context_) {
279 io_context_->stop();
280 }
281
282 // Join all worker threads
283 for (auto& thread : worker_threads_) {
284 if (thread.joinable()) {
285 thread.join();
286 }
287 }
288 worker_threads_.clear();
289
290 io_context_.reset();
291 }
