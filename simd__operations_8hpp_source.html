30 /**
31 * @brief SIMD-accelerated dot product
32 * @param a First vector
33 * @param b Second vector
34 * @return Dot product result
35 */
36 static T dot_product(std::span<const T> a, std::span<const T> b) {
37 if (a.size() != b.size()) {
38 throw std::invalid_argument("Vectors must have same size");
39 }
40
41 const std::size_t size = a.size();
42 const std::size_t simd_end = size - (size % simd_size);
43 T result = 0;
44
45 // Process aligned data in SIMD batches
46 for (std::size_t i = 0; i < simd_end; i += simd_size) {
47 auto va = batch_type::load_unaligned(&a[i]);
48 auto vb = batch_type::load_unaligned(&b[i]);
49 result += xsimd::reduce_add(va * vb);
50 }
51
52 // Process remaining elements
53 for (std::size_t i = simd_end; i < size; ++i) {
54 result += a[i] * b[i];
55 }
56
57 return result;
58 }

114 /**
115 * @brief SIMD mean and variance calculation
116 * @param data Input data
117 * @return Pair of (mean, variance)
118 */
119 static std::pair<double, double> mean_variance(std::span<const T> data) {
120 if (data.empty()) return {0.0, 0.0};
121
122 const std::size_t n = data.size();
123 const std::size_t simd_end = n - (n % simd_size);
124
125 batch_type sum_vec = batch_type(T(0));
126 batch_type sum_sq_vec = batch_type(T(0));
127
128 for (std::size_t i = 0; i < simd_end; i += simd_size) {
129 auto v = batch_type::load_unaligned(&data[i]);
130 sum_vec += v;
131 sum_sq_vec = xsimd::fma(v, v, sum_sq_vec);
132 }
133
134 T sum = xsimd::reduce_add(sum_vec);
135 T sum_sq = xsimd::reduce_add(sum_sq_vec);
136
137 // Process remaining elements
138 for (std::size_t i = simd_end; i < n; ++i) {
139 sum += data[i];
140 sum_sq += data[i] * data[i];
141 }
142
143 double mean = static_cast<double>(sum) / n;
144 double variance = (static_cast<double>(sum_sq) / n) - (mean * mean);
145
146 // Use Bessel's correction for sample variance
147 if (n > 1) {
148 variance = variance * n / (n - 1);
149 }
150
151 return {mean, variance};
152 }

159 static std::pair<T, T> minmax(std::span<const T> data) {
160 if (data.empty()) {
161 return {std::numeric_limits<T>::max(), std::numeric_limits<T>::lowest()};
162 }

165 const std::size_t simd_end = n - (n % simd_size);

168 batch_type max_vec = batch_type(std::numeric_limits<T>::lowest());

172 min_vec = xsimd::min(min_vec, v);
173 max_vec = xsimd::max(max_vec, v);
174 }
175
176 T min_val = xsimd::reduce_min(min_vec);
177 T max_val = xsimd::reduce_max(max_vec);
178
179 // Process remaining elements
180 for (std::size_t i = simd_end; i < n; ++i) {
181 min_val = std::min(min_val, data[i]);
182 max_val = std::max(max_val, data[i]);
183 }

243 * @brief SIMD-accelerated histogram computation
244 * @param data Input data
245 * @param bins Number of bins
246 * @param min Minimum value
247 * @param max Maximum value
248 * @return Histogram counts
249 */
250 static std::vector<std::size_t> fast_histogram(std::span<const T> data, std::size_t bins, T min,
251 T max) {
252 std::vector<std::size_t> hist(bins, 0);
253 if (data.empty() || bins == 0 || min >= max) return hist;
254
255 T range = max - min;
256 T bin_width = range / bins;
257 T inv_bin_width = T(1) / bin_width;
258
259 // Process data
260 for (const auto& value : data) {
261 if (value >= min && value <= max) {
262 std::size_t bin = static_cast<std::size_t>((value - min) * inv_bin_width);
263 if (bin >= bins) bin = bins - 1;
264 hist[bin]++;
265 }
266 }
267
268 return hist;
269 }

284 batch_type sum_batch(T(0));

286
287 // SIMD portion

289 auto m = batch_type::load_unaligned(&row[j]);
290 auto v = batch_type::load_unaligned(&vec[j]);
291 sum_batch = xsimd::fma(m, v, sum_batch);
292 }

295
296 // Scalar remainder
297 for (std::size_t j = simd_cols; j < cols; ++j) {
298 sum += row[j] * vec[j];

335 /**
336 * @brief Batch calculate probabilities for multiple scenarios
337 * @param total_dice Vector of total dice counts
338 * @param k_values Vector of minimum dice requirements
339 * @param face_values Vector of face values
340 * @return Vector of probabilities
341 */
342 static std::vector<T> batch_probability(std::span<const int> total_dice,
343 std::span<const int> k_values,
344 std::span<const int> face_values) {
345 if (total_dice.size() != k_values.size() || total_dice.size() != face_values.size()) {
346 throw std::invalid_argument("Input vectors must have same size");
347 }
348
349 std::vector<T> results(total_dice.size());
350
351 // Process in batches for better cache utilization
352 constexpr std::size_t BATCH_SIZE = 64;
353 for (std::size_t i = 0; i < total_dice.size(); i += BATCH_SIZE) {
354 std::size_t batch_end = std::min(i + BATCH_SIZE, total_dice.size());
355
356 for (std::size_t j = i; j < batch_end; ++j) {
357 results[j] = calculate_probability(total_dice[j], k_values[j], face_values[j]);
358 }
359 }
360
361 return results;
362 }
363
364 private:
365 static T calculate_probability(int total, int k, int face) {
366 if (k > total || k < 0) return T(0);
367 if (k == 0) return T(1);
368
369 T p = (face == 1) ? T(1.0 / 3.0) : T(1.0 / 6.0);
370 T q = T(1) - p;
371 T prob = T(0);
372
373 // Use logarithms for numerical stability with large values
374 if (total > 20) {
375 for (int i = k; i <= total; ++i) {
376 T log_term = log_binomial(total, i) + i * std::log(p) + (total - i) * std::log(q);
377 prob += std::exp(log_term);
378 }
379 } else {
380 // Direct calculation for small values
381 for (int i = k; i <= total; ++i) {
382 prob += binomial_coeff(total, i) * std::pow(p, i) * std::pow(q, total - i);
383 }
384 }
385
386 return prob;
387 }
388
389 static T log_binomial(int n, int k) {
390 if (k > n - k) k = n - k;
391 if (k == 0) return T(0);
392
393 T result = T(0);
394 for (int i = 0; i < k; ++i) {
395 result += std::log(T(n - i)) - std::log(T(i + 1));
396 }
397 return result;
398 }
399
400 static T binomial_coeff(int n, int k) {
401 if (k > n - k) k = n - k;
402 if (k == 0) return T(1);
403
404 T result = T(1);
405 for (int i = 0; i < k; ++i) {
406 result *= T(n - i) / T(i + 1);
407 }
408 return result;
409 }
