105 std::vector<T> result;
106 for (auto it = buffer_.rbegin(); it != buffer_.rend(); ++it) {

115 /**
116 * @brief Calculate simple moving average
117 * @param periods Number of periods
118 * @return Vector of moving averages
119 */
120 [[nodiscard]] std::vector<double> simple_moving_average(std::size_t periods) const {
121 if (buffer_.size() < periods) return {};
122
123 std::vector<double> sma;
124 sma.reserve(buffer_.size() - periods + 1);
125
126 // Calculate first SMA
127 double sum = 0;
128 for (std::size_t i = 0; i < periods; ++i) {
129 sum += static_cast<double>(buffer_[i].value);
130 }
131 sma.push_back(sum / periods);
132
133 // Calculate remaining SMAs using sliding window
134 for (std::size_t i = periods; i < buffer_.size(); ++i) {
135 sum = sum - static_cast<double>(buffer_[i - periods].value)
136 + static_cast<double>(buffer_[i].value);
137 sma.push_back(sum / periods);
138 }
139
140 return sma;
141 }

143 /**
144 * @brief Calculate exponential moving average
145 * @param alpha Smoothing factor (0 < alpha < 1)
146 * @return Vector of EMAs
147 */
148 [[nodiscard]] std::vector<double> exponential_moving_average(double alpha) const {
149 if (buffer_.empty() || alpha <= 0 || alpha >= 1) return {};
150
151 std::vector<double> ema;
152 ema.reserve(buffer_.size());
153
154 // First value is just the data point
155 ema.push_back(static_cast<double>(buffer_[0].value));
156
157 // Calculate EMA for remaining points
158 for (std::size_t i = 1; i < buffer_.size(); ++i) {
159 double new_ema = alpha * static_cast<double>(buffer_[i].value) + (1 - alpha) * ema.back();
160 ema.push_back(new_ema);

166 /**
167 * @brief Detect trend using linear regression
168 * @return Pair of (slope, intercept)
169 */
170 [[nodiscard]] std::pair<double, double> linear_trend() const {
171 if (buffer_.size() < 2) return {0.0, 0.0};
172
173 // Convert timestamps to numeric x values
174 std::vector<double> x_values;
175 std::vector<double> y_values;
176
177 auto first_time = buffer_.front().timestamp;
178
179 for (const auto& point : buffer_) {
180 auto duration
181 = std::chrono::duration_cast<std::chrono::seconds>(point.timestamp - first_time);
182 x_values.push_back(duration.count());
183 y_values.push_back(static_cast<double>(point.value));
184 }
185
186 // Calculate linear regression
187 double n = x_values.size();
188 double sum_x = std::accumulate(x_values.begin(), x_values.end(), 0.0);
189 double sum_y = std::accumulate(y_values.begin(), y_values.end(), 0.0);
190
191 double sum_xy = 0, sum_xx = 0;
192 for (std::size_t i = 0; i < n; ++i) {
193 sum_xy += x_values[i] * y_values[i];
194 sum_xx += x_values[i] * x_values[i];
195 }
196
197 double slope = (n * sum_xy - sum_x * sum_y) / (n * sum_xx - sum_x * sum_x);
198 double intercept = (sum_y - slope * sum_x) / n;
199
200 return {slope, intercept};
201 }

242 std::vector<std::size_t> outliers;
243 for (std::size_t i = 0; i < buffer_.size(); ++i) {

269 /**
270 * @brief Calculate autocorrelation
271 * @param lag Lag value
272 * @return Autocorrelation coefficient
273 */
274 [[nodiscard]] boost::optional<double> autocorrelation(std::size_t lag) const {
275 if (lag >= buffer_.size()) return boost::none;
276
277 // Calculate mean
278 double sum = 0;
279 for (const auto& point : buffer_) {
280 sum += static_cast<double>(point.value);
281 }
282 double mean = sum / buffer_.size();
283
284 // Calculate autocorrelation
285 double numerator = 0;
286 double denominator = 0;
287
288 for (std::size_t i = 0; i < buffer_.size() - lag; ++i) {
289 double x1 = static_cast<double>(buffer_[i].value) - mean;
290 double x2 = static_cast<double>(buffer_[i + lag].value) - mean;
291 numerator += x1 * x2;
292 }
293
294 for (const auto& point : buffer_) {
295 double diff = static_cast<double>(point.value) - mean;
296 denominator += diff * diff;
297 }
298
299 if (denominator == 0) return boost::none;
300
301 return numerator / denominator;
302 }

317 /**
318 * @brief Resample time series to fixed intervals
319 * @param interval Target interval duration
320 * @return Resampled time series
321 */
322 [[nodiscard]] TimeSeries resample(duration interval) const {
323 TimeSeries resampled;
324 if (buffer_.empty()) return resampled;
325
326 auto current_time = buffer_.front().timestamp;
327 auto end_time = buffer_.back().timestamp;
328
329 while (current_time <= end_time) {
330 // Find nearest point
331 auto it = std::min_element(buffer_.begin(), buffer_.end(),
332 [current_time](const auto& a, const auto& b) {
333 return std::abs((a.timestamp - current_time).count())
334 < std::abs((b.timestamp - current_time).count());
335 });
336
337 if (it != buffer_.end()) {
338 resampled.add(current_time, it->value);
339 }
340
341 current_time += interval;
342 }
343
344 return resampled;
345 }
