87 /**
88 * @brief Prepare a statement with caching and LRU eviction
89 * @param sql SQL statement to prepare
90 * @return Prepared statement or error
91 */
92 DatabaseResult<std::shared_ptr<PreparedStatement>> prepare(const std::string& sql) {
93 boost::lock_guard<boost::mutex> lock(cache_mutex_);
94
95 // Check cache first
96 auto it = statement_cache_.find(sql);
97 if (it != statement_cache_.end()) {
98 // Update LRU position
99 auto lru_it = cache_lru_map_[sql];
100 cache_lru_list_.erase(lru_it);
101 cache_lru_list_.push_front(sql);
102 cache_lru_map_[sql] = cache_lru_list_.begin();
103
104 // Reset and return cached statement
105 it->second->reset();
106 it->second->clear_bindings();
107 BOOST_LOG_TRIVIAL(trace) << "Using cached prepared statement";
108 return it->second;
109 }
110
111 // Check cache size limits before adding new statement
112 if (statement_cache_.size() >= MAX_CACHED_STATEMENTS) {
113 // Evict least recently used statement
114 std::string lru_key = cache_lru_list_.back();
115 cache_lru_list_.pop_back();
116 cache_lru_map_.erase(lru_key);
117 statement_cache_.erase(lru_key);
118 BOOST_LOG_TRIVIAL(trace) << "Evicted LRU statement from cache";
119 }
120
121 // Check estimated memory usage
122 CacheStats stats = get_cache_stats();
123 if (stats.cache_memory_estimate > MAX_CACHE_MEMORY_MB * 1024 * 1024) {
124 // Clear half of the cache if memory limit exceeded
125 size_t to_remove = statement_cache_.size() / 2;
126 for (size_t i = 0; i < to_remove && !cache_lru_list_.empty(); ++i) {
127 std::string lru_key = cache_lru_list_.back();
128 cache_lru_list_.pop_back();
129 cache_lru_map_.erase(lru_key);
130 statement_cache_.erase(lru_key);
131 }
132 BOOST_LOG_TRIVIAL(warning)
133 << "Cache memory limit exceeded, cleared " << to_remove << " statements";
134 }
135
136 // Prepare new statement
137 try {
138 auto& conn_mgr = ConnectionManager::instance();
139 auto conn = conn_mgr.acquire_connection();
140
141 if (!conn) {
142 return DatabaseError(DatabaseErrorType::ConnectionFailed,
143 "Failed to acquire database connection");
144 }
145
146 sqlite3_stmt* stmt_raw = nullptr;
147 int result = sqlite3_prepare_v2(conn->get(), sql.c_str(), -1, &stmt_raw, nullptr);
148
149 if (result != SQLITE_OK) {
150 std::string error_msg = conn->get_last_error();
151 BOOST_LOG_TRIVIAL(error)
152 << "Failed to prepare statement: " << sql << " Error: " << error_msg;
153 return DatabaseError(DatabaseErrorType::PreparedStatementFailed,
154 "Failed to prepare statement: " + error_msg);
155 }
156
157 auto stmt = std::make_shared<PreparedStatement>(stmt_raw, sql);
158
159 // Cache the statement with LRU tracking
160 statement_cache_[sql] = stmt;
161 cache_lru_list_.push_front(sql);
162 cache_lru_map_[sql] = cache_lru_list_.begin();
163
164 BOOST_LOG_TRIVIAL(debug) << "Prepared and cached statement: " << sql
165 << " (cache size: " << statement_cache_.size() << ")";
166
167 return stmt;
168
169 } catch (const std::exception& e) {
170 BOOST_LOG_TRIVIAL(error) << "Exception in prepare: " << e.what();
171 return DatabaseError(DatabaseErrorType::InternalError, e.what());
172 }
173 }

175 /**
176 * @brief Execute a prepared statement
177 * @param stmt Prepared statement
178 * @param callback Optional callback for each row
179 * @return Result or error
180 */
181 DatabaseResult<int> execute_prepared(std::shared_ptr<PreparedStatement> stmt,
182 QueryCallback callback = nullptr) {
183 if (!stmt || !*stmt) {
184 return DatabaseError(DatabaseErrorType::InvalidParameter, "Invalid prepared statement");
185 }
186
187 auto start_time = std::chrono::high_resolution_clock::now();
188 int rows_affected = 0;
189
190 try {
191 int result;
192 while ((result = stmt->step()) == SQLITE_ROW) {
193 if (callback) {
194 callback(*stmt);
195 }
196 rows_affected++;
197 }
198
199 if (result != SQLITE_DONE) {
200 auto& conn_mgr = ConnectionManager::instance();
201 auto conn = conn_mgr.acquire_connection();
202 std::string error_msg = conn ? conn->get_last_error() : "Unknown error";
203
204 BOOST_LOG_TRIVIAL(error) << "Statement execution failed: " << error_msg;
205 return DatabaseError(DatabaseErrorType::QueryFailed,
206 "Statement execution failed: " + error_msg);
207 }
208
209 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(
210 std::chrono::high_resolution_clock::now() - start_time);
211
212 BOOST_LOG_TRIVIAL(trace) << "Prepared statement executed in " << duration.count() << "μs, "
213 << rows_affected << " rows affected";
214
215 return rows_affected;
216
217 } catch (const std::exception& e) {
218 BOOST_LOG_TRIVIAL(error) << "Exception in execute_prepared: " << e.what();

305 /**
306 * @brief Execute a function within a transaction with automatic rollback on failure
307 * @param func Function to execute
308 * @return Result or error
309 */
310 DatabaseResult<void> with_transaction(TransactionFunc func) {
311 // Begin transaction
312 auto begin_result = begin_transaction();
313 if (!begin_result) {
314 return begin_result;
315 }
316
317 // Use scope_exit for automatic rollback
318 bool should_rollback = true;
319 BOOST_SCOPE_EXIT(&should_rollback, this_) {
320 if (should_rollback && this_->in_transaction_) {
321 this_->rollback_transaction();
322 }
323 }
324 BOOST_SCOPE_EXIT_END
325
326 try {
327 // Execute the function
328 auto result = func(*this);
329
330 if (result) {
331 // Success - commit
332 auto commit_result = commit_transaction();
333 if (commit_result) {
334 should_rollback = false;
335 return DatabaseResult<void>();
336 }
337 return commit_result;
338 } else {
339 // Function returned error
340 BOOST_LOG_TRIVIAL(warning)
341 << "Transaction function failed: " << result.error().full_message();
342 return result;
343 }
344
345 } catch (const std::exception& e) {
346 BOOST_LOG_TRIVIAL(error) << "Exception in transaction: " << e.what();
347 return DatabaseError(DatabaseErrorType::TransactionFailed,
348 "Exception in transaction: " + std::string(e.what()));
349 }
