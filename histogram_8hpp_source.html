180 /**
181 * @brief Find mode (bin with highest count)
182 * @return Pair of (bin_center, count)
183 */
184 [[nodiscard]] std::pair<T, double> mode() const {
185 if (hist_.size() == 0) {
186 return {T{}, 0.0};
187 }
188
189 auto max_it = std::max_element(hist_.begin(), hist_.end());
190 if (max_it == hist_.end()) {
191 return {T{}, 0.0};
192 }
193
194 std::size_t max_idx = std::distance(hist_.begin(), max_it);
195 const auto& axis = hist_.axis();
196
197 // Bounds check
198 if (max_idx >= axis.size()) {
199 return {T{}, 0.0};
200 }
201
202 T center;
203 try {
204 if constexpr (std::is_integral_v<T>) {
205 auto bin = axis.bin(max_idx);
206 center = static_cast<T>((bin.lower() + bin.upper()) / 2);
207 } else {
208 center = axis.bin(max_idx).center();
209 }
210 } catch (...) {
211 // Fallback: return middle of histogram range
212 center = static_cast<T>((min_value_ + max_value_) / 2);
213 }
214
215 return {center, *max_it};
216 }

223 if (total_count_ == 0) return 0.0;
224
225 double sum = 0.0;
226 const auto& axis = hist_.axis();
227
228 for (std::size_t i = 0; i < hist_.size(); ++i) {
229 T center;
230 if constexpr (std::is_integral_v<T>) {
231 auto bin = axis.bin(i);
232 center = (bin.lower() + bin.upper()) / 2;
233 } else {
234 center = axis.bin(i).center();

236 sum += center * hist_[i];

249 double m = mean();
250 double sum_sq = 0.0;
251 const auto& axis = hist_.axis();
252
253 for (std::size_t i = 0; i < hist_.size(); ++i) {
254 T center;

286 const auto& axis = hist_.axis();
287
288 for (std::size_t i = 0; i < hist_.size(); ++i) {
289 cumsum += hist_[i];
290 if (cumsum >= target) {
291 try {
292 if constexpr (std::is_integral_v<T>) {
293 auto bin = axis.bin(i);
294 return static_cast<T>((bin.lower() + bin.upper()) / 2);

362 class DiceHistogram : public Histogram<int> {
363 public:
364 DiceHistogram() : Histogram<int>(6, 1, 7) {} // 6 bins for faces 1-6

366 /**
367 * @brief Check if dice distribution is fair
368 * @param alpha Significance level
369 * @return True if distribution appears uniform
370 */
371 [[nodiscard]] bool is_fair(double alpha = 0.05) const {
372 // Chi-square test for uniformity
373 double expected = total_count() / 6.0;
374 double chi_square = 0.0;
375
376 for (int face = 1; face <= 6; ++face) {
377 double observed = count_at(face);
378 double diff = observed - expected;
379 chi_square += (diff * diff) / expected;
380 }
381
382 // Critical value for 5 degrees of freedom
383 double critical_value = 11.070; // alpha = 0.05
384 return chi_square < critical_value;
385 }
