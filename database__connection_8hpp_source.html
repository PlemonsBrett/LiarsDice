150 /**
151 * @brief Check connection health
152 * @return true if connection is healthy
153 */
154 bool check_health() const {
155 if (!connection_) {
156 return false;
157 }
158
159 // Quick integrity check
160 sqlite3_stmt* stmt = nullptr;
161 int result = sqlite3_prepare_v2(connection_.get(), "PRAGMA quick_check", -1, &stmt, nullptr);
162
163 if (result != SQLITE_OK) {
164 return false;
165 }
166
167 bool is_healthy = false;
168 if (sqlite3_step(stmt) == SQLITE_ROW) {
169 const char* status = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 0));
170 is_healthy = (status && std::string(status) == "ok");
171 }
172
173 sqlite3_finalize(stmt);
174 return is_healthy;
175 }

331 void configure_connection() {
332 if (!connection_) return;
333
334 // Execute pragmas directly without locking again
335 char* error_msg = nullptr;
336
337 // Set pragmas for performance and safety
338 const char* pragmas[] = {
339 "PRAGMA journal_mode=WAL", "PRAGMA synchronous=NORMAL", "PRAGMA foreign_keys=ON",
340 "PRAGMA cache_size=-64000", // 64MB cache
341 "PRAGMA temp_store=MEMORY",
342 "PRAGMA mmap_size=268435456" // 256MB mmap
343 };
344
345 for (const char* pragma : pragmas) {
346 int result = sqlite3_exec(connection_.get(), pragma, nullptr, nullptr, &error_msg);
347 if (result != SQLITE_OK && error_msg) {
348 sqlite3_free(error_msg);
349 }
350 }
351
352 // Set busy timeout directly
353 sqlite3_busy_timeout(connection_.get(), 5000); // 5 seconds
354 }
