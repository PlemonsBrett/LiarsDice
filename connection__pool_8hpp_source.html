135 /**
136 * @brief Acquire a connection from the pool
137 * @param timeout Maximum time to wait for connection
138 * @return Pooled connection that auto-returns on destruction
139 */
140 PooledConnection acquire(std::chrono::milliseconds timeout = std::chrono::milliseconds(5000)) {
141 boost::unique_lock<boost::mutex> lock(mutex_);
142
143 if (!is_running_) {
144 return PooledConnection();
145 }
146
147 // Wait for available connection or create new one
148 if (!condition_.wait_for(lock, boost::chrono::milliseconds(timeout.count()), [this] {
149 return !available_connections_.empty() || total_connections_ < config_.max_connections;
150 })) {
151 return PooledConnection(); // Timeout
152 }
153
154 ConnectionPtr conn;
155
156 // Try to get existing connection
157 if (!available_connections_.empty()) {
158 conn = available_connections_.front();
159 available_connections_.pop();
160
161 // Verify connection is still good
162 if (!conn->is_open() || !verify_connection(conn)) {
163 conn.reset();
164 }
165 }
166
167 // Create new connection if needed
168 if (!conn && total_connections_ < config_.max_connections) {
169 conn = create_connection();
170 }
171
172 if (!conn) {
173 return PooledConnection();
174 }
175
176 // Return pooled connection with auto-return function
177 return PooledConnection(conn, [this](ConnectionPtr c) { return_connection(c); });
178 }

253 */
254 void return_connection(ConnectionPtr conn) {
255 boost::lock_guard<boost::mutex> lock(mutex_);
256
257 if (!is_running_) {
258 return;
259 }
260
261 // Check if connection is still healthy
262 if (conn && conn->is_open() && verify_connection(conn)) {
263 // Check idle timeout
264 if (conn->get_idle_time() < config_.idle_timeout) {
265 available_connections_.push(conn);
266 condition_.notify_one();
267 return;
268 }
269 }
270
271 // Connection is bad or too old, close it
272 if (conn) {
273 conn->close();
274 --total_connections_;
275 }
276
277 // Maintain minimum connections
278 if (total_connections_ < config_.min_connections) {
279 auto new_conn = create_connection();
280 if (new_conn && new_conn->is_open()) {
281 available_connections_.push(new_conn);

301 /**
302 * @brief Perform periodic health checks

304 void perform_health_check() {
305 boost::lock_guard<boost::mutex> lock(mutex_);
306
307 if (!is_running_) {
308 return;
309 }
310
311 std::queue<ConnectionPtr> healthy_connections;
312
313 // Check all available connections
314 while (!available_connections_.empty()) {
315 auto conn = available_connections_.front();
316 available_connections_.pop();
317
318 if (verify_connection(conn) && conn->get_idle_time() < config_.idle_timeout) {
319 healthy_connections.push(conn);
320 } else {
321 conn->close();
322 --total_connections_;
323 }
324 }
325
326 // Put healthy connections back
327 available_connections_ = std::move(healthy_connections);
328
329 // Ensure minimum connections
330 while (available_connections_.size() < config_.min_connections
331 && total_connections_ < config_.max_connections) {
332 auto conn = create_connection();
333 if (conn && conn->is_open()) {
334 available_connections_.push(conn);
335 }
336 }
337
338 condition_.notify_all();
339 }
