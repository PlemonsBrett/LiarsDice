204 /**
205 * @brief Check if distribution is approximately normal
206 * @param alpha Significance level (default: 0.05)
207 * @return True if skewness and kurtosis suggest normality
208 */
209 [[nodiscard]] bool is_normal_distributed(double alpha = 0.05) const {
210 // Simple normality check based on skewness and excess kurtosis
211 // For normal distribution: skewness ≈ 0, excess kurtosis ≈ 0
212 T skew = std::abs(skewness());
213 T excess_kurt = std::abs(kurtosis() - 3.0); // kurtosis - 3 for excess
214
215 // Rule of thumb thresholds
216 double skew_threshold = 2.0;
217 double kurt_threshold = 7.0;
218
219 return skew < skew_threshold && excess_kurt < kurt_threshold;
220 }

261 * @brief Get probability of specific face value
262 * @param face Face value (1-6)
263 * @return Empirical probability
264 */
265 [[nodiscard]] double face_probability(unsigned int face) const {
266 if (face < 1 || face > 6) return 0.0;
267
268 // Calculate total dice count
269 unsigned int total_dice = 0;
270 for (auto count : face_counts_) {
271 total_dice += count;
272 }
273
274 if (total_dice == 0) return 0.0;
275
276 return static_cast<double>(face_counts_[face - 1]) / total_dice;
277 }

279 /**
280 * @brief Get chi-square test statistic for uniformity
281 * @return Chi-square value
282 */
283 [[nodiscard]] double chi_square_uniformity() const {
284 // Calculate total dice count
285 unsigned int total_dice = 0;
286 for (auto count : face_counts_) {
287 total_dice += count;
288 }
289
290 if (total_dice == 0) return 0.0;
291
292 double expected = total_dice / 6.0; // Expected count per face for uniform distribution
293 double chi_square = 0.0;
294
295 for (auto count : face_counts_) {
296 double diff = count - expected;
297 chi_square += (diff * diff) / expected;
298 }
299
300 return chi_square;
301 }
