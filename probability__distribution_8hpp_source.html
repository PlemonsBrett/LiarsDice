20 /**
21 * @brief Base interface for probability distributions
22 *
23 * Provides unified interface for various boost::math distributions
24 * with support for PDF, CDF, quantiles, and random sampling.
25 */
26 class IProbabilityDistribution {
27 public:
28 virtual ~IProbabilityDistribution() = default;
29
30 /**
31 * @brief Probability density function
32 * @param x Point to evaluate
33 * @return PDF value at x
34 */
35 [[nodiscard]] virtual double pdf(double x) const = 0;
36
37 /**
38 * @brief Cumulative distribution function
39 * @param x Point to evaluate
40 * @return CDF value at x
41 */
42 [[nodiscard]] virtual double cdf(double x) const = 0;
43
44 /**
45 * @brief Quantile function (inverse CDF)
46 * @param p Probability (0 < p < 1)
47 * @return Quantile value
48 */
49 [[nodiscard]] virtual double quantile(double p) const = 0;
50
51 /**
52 * @brief Mean of the distribution
53 * @return Mean value
54 */
55 [[nodiscard]] virtual double mean() const = 0;

57 /**
58 * @brief Variance of the distribution
59 * @return Variance

61 [[nodiscard]] virtual double variance() const = 0;

64 * @brief Standard deviation
65 * @return Standard deviation
66 */
67 [[nodiscard]] virtual double standard_deviation() const { return std::sqrt(variance()); }

69 /**
70 * @brief Generate random sample
71 * @param gen Random number generator
72 * @return Random value from distribution
73 */
74 virtual double sample(br::mt19937& gen) const = 0;
75
76 /**
77 * @brief Generate multiple random samples
78 * @param gen Random number generator
79 * @param n Number of samples
80 * @return Vector of samples
81 */
82 virtual std::vector<double> sample(br::mt19937& gen, std::size_t n) const {
83 std::vector<double> samples;
84 samples.reserve(n);
85 for (std::size_t i = 0; i < n; ++i) {
86 samples.push_back(sample(gen));
87 }
88 return samples;
89 }

92 * @brief Get distribution name
93 * @return String name of distribution

217 /**
218 * @brief Beta distribution
219 */
220 class BetaDistribution : public DistributionWrapper<bm::beta_distribution<>> {
221 public:
222 BetaDistribution(double alpha, double beta)
223 : DistributionWrapper(alpha, beta), alpha_(alpha), beta_(beta) {
224 // Beta sampling using gamma distributions
225 gamma_alpha_ = std::make_unique<br::gamma_distribution<>>(alpha, 1.0);
226 gamma_beta_ = std::make_unique<br::gamma_distribution<>>(beta, 1.0);
227 }
228
229 double sample(br::mt19937& gen) const override {
230 // Beta = Gamma(α,1) / (Gamma(α,1) + Gamma(β,1))
231 double x = (*gamma_alpha_)(gen);
232 double y = (*gamma_beta_)(gen);
233 return x / (x + y);
234 }
235
236 [[nodiscard]] std::string name() const override { return "Beta"; }
237
238 [[nodiscard]] double alpha() const { return alpha_; }
239 [[nodiscard]] double beta() const { return beta_; }
240
241 private:
242 double alpha_, beta_;
243 mutable std::unique_ptr<br::gamma_distribution<>> gamma_alpha_;
244 mutable std::unique_ptr<br::gamma_distribution<>> gamma_beta_;
245 };

297 /**
298 * @brief Statistical hypothesis testing utilities
299 */
300 class HypothesisTest {
301 public:
302 /**
303 * @brief Perform Kolmogorov-Smirnov test
304 * @param data Sample data
305 * @param dist Distribution to test against
306 * @return Pair of (test_statistic, p_value)
307 */
308 static std::pair<double, double> kolmogorov_smirnov_test(const std::vector<double>& data,
309 const IProbabilityDistribution& dist) {
310 if (data.empty()) return {0.0, 1.0};
311
312 // Sort data for empirical CDF
313 std::vector<double> sorted_data = data;
314 std::sort(sorted_data.begin(), sorted_data.end());
315
316 // Calculate K-S statistic
317 double max_diff = 0.0;
318 std::size_t n = sorted_data.size();
319
320 for (std::size_t i = 0; i < n; ++i) {
321 double empirical_cdf = (i + 1.0) / n;
322 double theoretical_cdf = dist.cdf(sorted_data[i]);
323 double diff1 = std::abs(empirical_cdf - theoretical_cdf);
324 double diff2 = std::abs(i / static_cast<double>(n) - theoretical_cdf);
325 max_diff = std::max({max_diff, diff1, diff2});
326 }
327
328 // Approximate p-value (simplified)
329 double sqrt_n = std::sqrt(static_cast<double>(n));
330 double p_value = 2.0 * std::exp(-2.0 * n * max_diff * max_diff);
331
332 return {max_diff, p_value};
333 }
334
335 /**
336 * @brief Chi-square goodness of fit test
337 * @param observed Observed frequencies
338 * @param expected Expected frequencies
339 * @return Pair of (chi_square_statistic, degrees_of_freedom)
340 */
341 static std::pair<double, int> chi_square_test(const std::vector<double>& observed,
342 const std::vector<double>& expected) {
343 if (observed.size() != expected.size() || observed.empty()) {
344 return {0.0, 0};
345 }
346
347 double chi_square = 0.0;
348 for (std::size_t i = 0; i < observed.size(); ++i) {
349 if (expected[i] > 0) {
350 double diff = observed[i] - expected[i];
351 chi_square += (diff * diff) / expected[i];
352 }
353 }
354
355 int df = static_cast<int>(observed.size()) - 1;
356 return {chi_square, df};
357 }
358 };

368 * @param successes Number of successes

380 /**
381 * @brief Calculate credible interval for beta distribution
382 * @param dist Beta distribution
383 * @param credibility Credibility level (e.g., 0.95)
384 * @return Pair of (lower_bound, upper_bound)
385 */
386 static std::pair<double, double> credible_interval(const BetaDistribution& dist,
387 double credibility = 0.95) {
388 double alpha = (1.0 - credibility) / 2.0;
389 return {dist.quantile(alpha), dist.quantile(1.0 - alpha)};
390 }
