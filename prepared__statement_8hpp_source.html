33 bool bind(int index, const ValueType& value) {
34 if (!stmt_) return false;
35
36 return std::visit(
37 [this, index](auto&& arg) -> bool {
38 using T = std::decay_t<decltype(arg)>;
39
40 if constexpr (std::is_same_v<T, std::monostate>) {
41 return sqlite3_bind_null(stmt_.get(), index) == SQLITE_OK;
42 } else if constexpr (std::is_same_v<T, int>) {
43 return sqlite3_bind_int(stmt_.get(), index, arg) == SQLITE_OK;
44 } else if constexpr (std::is_same_v<T, int64_t>) {
45 return sqlite3_bind_int64(stmt_.get(), index, arg) == SQLITE_OK;
46 } else if constexpr (std::is_same_v<T, double>) {
47 return sqlite3_bind_double(stmt_.get(), index, arg) == SQLITE_OK;
48 } else if constexpr (std::is_same_v<T, std::string>) {
49 return sqlite3_bind_text(stmt_.get(), index, arg.c_str(),
50 static_cast<int>(arg.size()), SQLITE_TRANSIENT)
51 == SQLITE_OK;
52 } else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
53 return sqlite3_bind_blob(stmt_.get(), index, arg.data(), static_cast<int>(arg.size()),
54 SQLITE_TRANSIENT)

56 }
57 return false;

112 case SQLITE_BLOB: {

117 return std::vector<uint8_t>(data, data + size);
