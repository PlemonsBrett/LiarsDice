111 /**
112 * @brief Acquire object from pool
113 * @tparam Args Constructor arguments
114 * @param args Arguments to forward to constructor
115 * @return Unique pointer to object with custom deleter

117 template <typename... Args>
118 std::unique_ptr<T, std::function<void(T*)>> acquire(Args&&... args) {

158 /**
159 * @brief Memory arena for temporary allocations
160 *
161 * Stack-based allocation pattern for temporary objects
162 * with bulk deallocation.
163 */
164 class MemoryArena {
165 public:
166 explicit MemoryArena(std::size_t size = 1024 * 1024) // 1MB default
167 : buffer_(size), offset_(0) {}
168
169 /**
170 * @brief Allocate memory from arena
171 * @param size Number of bytes
172 * @param alignment Alignment requirement
173 * @return Pointer to allocated memory
174 */
175 void* allocate(std::size_t size, std::size_t alignment = alignof(std::max_align_t)) {
176 // Align offset
177 std::size_t aligned_offset = (offset_ + alignment - 1) & ~(alignment - 1);
178
179 if (aligned_offset + size > buffer_.size()) {
180 throw std::bad_alloc();
181 }
182
183 void* ptr = buffer_.data() + aligned_offset;
184 offset_ = aligned_offset + size;
185
186 return ptr;
187 }

263 void track_allocation(std::size_t size) {
264 stats_.allocations++;
265 stats_.bytes_allocated += size;
266 stats_.current_usage += size;

268 // Update peak usage
269 std::size_t current = stats_.current_usage.load();
270 std::size_t peak = stats_.peak_usage.load();
271 while (current > peak && !stats_.peak_usage.compare_exchange_weak(peak, current)) {
272 // Loop until successful
273 }
274 }

277 stats_.deallocations++;
278 stats_.bytes_deallocated += size;
279 stats_.current_usage -= size;
280 }
