99 /**
100 * @brief Calculate statistics over the buffer
101 * @tparam Extractor Function to extract numeric value from T
102 * @param extractor Function to extract value for statistics
103 * @return Tuple of (mean, std_dev, min, max)
104 */
105 template <typename Extractor> [[nodiscard]] auto calculate_statistics(Extractor extractor) const
106 -> std::tuple<double, double, double, double> {
107 if (buffer_.empty()) {
108 return {0.0, 0.0, 0.0, 0.0};
109 }
110
111 std::vector<double> values;
112 values.reserve(buffer_.size());
113
114 for (const auto& elem : buffer_) {
115 values.push_back(static_cast<double>(extractor(elem)));
116 }
117
118 double sum = std::accumulate(values.begin(), values.end(), 0.0);
119 double mean = sum / values.size();
120
121 double sq_sum = std::accumulate(
122 values.begin(), values.end(), 0.0,
123 [mean](double acc, double val) { return acc + (val - mean) * (val - mean); });
124 double std_dev = std::sqrt(sq_sum / values.size());
125
126 auto [min_it, max_it] = std::minmax_element(values.begin(), values.end());
127
128 return {mean, std_dev, *min_it, *max_it};
129 }

131 /**
132 * @brief Find patterns in the buffer
133 * @tparam Predicate Predicate type
134 * @param pattern_size Size of pattern to search for
135 * @param predicate Function to check if elements form a pattern
136 * @return Vector of starting indices where patterns are found
137 */
138 template <typename Predicate>
139 [[nodiscard]] std::vector<size_type> find_patterns(size_type pattern_size,
140 Predicate predicate) const {
141 std::vector<size_type> indices;

144
145 for (size_type i = 0; i <= buffer_.size() - pattern_size; ++i) {
146 if (predicate(buffer_.begin() + i, buffer_.begin() + i + pattern_size)) {
