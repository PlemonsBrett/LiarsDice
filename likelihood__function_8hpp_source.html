115 /**
116 * @brief Bernoulli likelihood for binary outcomes
117 *
118 * Models the probability of success in a single trial
119 */
120 template <typename T = double> class BernoulliLikelihood : public LikelihoodFunction<T> {
121 public:
122 using typename LikelihoodFunction<T>::vector_type;
123
124 T evaluate(T theta, T observation) const override {
125 if (theta < 0 || theta > 1) {
126 throw std::domain_error("Bernoulli parameter must be in [0,1]");
127 }
128
129 if (observation == 1) {
130 return theta;
131 } else if (observation == 0) {
132 return 1 - theta;
133 } else {
134 throw std::invalid_argument("Bernoulli observation must be 0 or 1");
135 }
136 }
137
138 T evaluate(T theta, const vector_type& observations) const override {
139 T likelihood = 1.0;
140 for (std::size_t i = 0; i < observations.size(); ++i) {
141 likelihood *= evaluate(theta, observations[i]);
142 }
143 return likelihood;
144 }
145
146 T log_evaluate(T theta, T observation) const override {
147 if (theta < 0 || theta > 1) {
148 throw std::domain_error("Bernoulli parameter must be in [0,1]");
149 }
150
151 if (observation == 1) {
152 return std::log(theta);
153 } else if (observation == 0) {
154 return std::log(1 - theta);
155 } else {
156 throw std::invalid_argument("Bernoulli observation must be 0 or 1");
157 }
158 }
159
160 T log_evaluate(T theta, const vector_type& observations) const override {
161 T log_likelihood = 0.0;
162 for (std::size_t i = 0; i < observations.size(); ++i) {
163 log_likelihood += log_evaluate(theta, observations[i]);
164 }
165 return log_likelihood;
166 }
167
168 vector_type sufficient_statistics(const vector_type& observations) const override {
169 vector_type stats(2);
170 stats[0] = std::accumulate(observations.begin(), observations.end(), T(0));
171 stats[1] = observations.size();
172 return stats;
173 }
174
175 std::string family() const override { return "bernoulli"; }
176
177 std::unique_ptr<LikelihoodFunction<T>> clone() const override {
178 return std::make_unique<BernoulliLikelihood<T>>();
179 }
180
181 bool has_conjugate_prior() const override { return true; }
182
183 std::optional<std::string> conjugate_prior_family() const override { return "beta"; }
184 };

186 /**
187 * @brief Binomial likelihood for count data
188 *
189 * Models the number of successes in a fixed number of trials
190 */
191 template <typename T = double> class BinomialLikelihood : public LikelihoodFunction<T> {
192 public:
193 using typename LikelihoodFunction<T>::vector_type;
194
195 explicit BinomialLikelihood(std::size_t n_trials) : n_trials_(n_trials) {}
196
197 T evaluate(T theta, T observation) const override {
198 if (theta < 0 || theta > 1) {
199 throw std::domain_error("Binomial parameter must be in [0,1]");
200 }
201
202 std::size_t k = static_cast<std::size_t>(observation);
203 if (k > n_trials_) {
204 throw std::invalid_argument("Observation exceeds number of trials");
205 }
206
207 boost::math::binomial_distribution<T> dist(n_trials_, theta);
208 return boost::math::pdf(dist, k);
209 }
210
211 T evaluate(T theta, const vector_type& observations) const override {
212 T likelihood = 1.0;
213 for (std::size_t i = 0; i < observations.size(); ++i) {
214 likelihood *= evaluate(theta, observations[i]);
215 }
216 return likelihood;
217 }
218
219 vector_type sufficient_statistics(const vector_type& observations) const override {
220 vector_type stats(2);
221 stats[0] = std::accumulate(observations.begin(), observations.end(), T(0));
222 stats[1] = observations.size() * n_trials_;
223 return stats;
224 }
225
226 std::string family() const override { return "binomial"; }
227
228 std::unique_ptr<LikelihoodFunction<T>> clone() const override {
229 return std::make_unique<BinomialLikelihood<T>>(n_trials_);
230 }
231
232 bool has_conjugate_prior() const override { return true; }
233
234 std::optional<std::string> conjugate_prior_family() const override { return "beta"; }
235
236 private:
237 std::size_t n_trials_;
238 };

241 * @brief Normal likelihood with known variance
242 *
243 * Models continuous data with Gaussian noise of known variance
244 */
245 template <typename T = double> class NormalKnownVarianceLikelihood
246 : public LikelihoodFunction<T> {
247 public:
248 using typename LikelihoodFunction<T>::vector_type;
249
250 explicit NormalKnownVarianceLikelihood(T variance)
251 : variance_(variance), std_dev_(std::sqrt(variance)) {
252 if (variance <= 0) {
253 throw std::invalid_argument("Variance must be positive");
254 }
255 }
256
257 T evaluate(T theta, T observation) const override {
258 boost::math::normal_distribution<T> dist(theta, std_dev_);
259 return boost::math::pdf(dist, observation);
260 }
261
262 T evaluate(T theta, const vector_type& observations) const override {
263 T likelihood = 1.0;
264 for (std::size_t i = 0; i < observations.size(); ++i) {
265 likelihood *= evaluate(theta, observations[i]);
266 }
267 return likelihood;
268 }
269
270 T log_evaluate(T theta, T observation) const override {
271 T z = (observation - theta) / std_dev_;
272 return -0.5 * std::log(2 * M_PI) - std::log(std_dev_) - 0.5 * z * z;
273 }
274
275 T log_evaluate(T theta, const vector_type& observations) const override {
276 T log_likelihood = 0.0;
277 for (std::size_t i = 0; i < observations.size(); ++i) {
278 log_likelihood += log_evaluate(theta, observations[i]);
279 }
280 return log_likelihood;
281 }
282
283 vector_type sufficient_statistics(const vector_type& observations) const override {
284 vector_type stats(2);

286 stats[1] = observations.size();
287 return stats;
288 }

291

394 /**
395 * @brief Factory function for creating common likelihood functions
396 */
397 template <typename T = double>
398 std::unique_ptr<LikelihoodFunction<T>> create_likelihood(const std::string& family_name,
399 const std::vector<T>& parameters = {}) {
400 if (family_name == "bernoulli") {
401 return std::make_unique<BernoulliLikelihood<T>>();
402 } else if (family_name == "binomial" && !parameters.empty()) {
403 return std::make_unique<BinomialLikelihood<T>>(static_cast<std::size_t>(parameters[0]));
404 } else if (family_name == "normal_known_variance" && !parameters.empty()) {
405 return std::make_unique<NormalKnownVarianceLikelihood<T>>(parameters[0]);
406 } else if (family_name == "poisson") {
407 return std::make_unique<PoissonLikelihood<T>>();
408 }
409
410 throw std::invalid_argument("Unknown likelihood family or missing parameters");
411 }
