115 /**
116 * @brief Beta distribution prior - conjugate for Bernoulli/Binomial likelihoods
117 *
118 * Particularly useful for modeling probabilities and proportions in [0,1]
119 */
120 template <typename T = double> class BetaPrior : public PriorDistribution<T> {
121 public:
122 using distribution_type = boost::math::beta_distribution<T>;
123
124 /**
125 * @brief Construct a Beta prior
126 * @param alpha Shape parameter α > 0
127 * @param beta Shape parameter β > 0
128 */
129 BetaPrior(T alpha, T beta) : dist_(alpha, beta), alpha_(alpha), beta_(beta) {
130 if (alpha <= 0 || beta <= 0) {
131 throw std::invalid_argument("Beta parameters must be positive");
132 }
133 }
134
135 T pdf(T x) const override {
136 if (x < 0 || x > 1) return 0;
137 return boost::math::pdf(dist_, x);
138 }
139
140 T log_pdf(T x) const override {
141 if (x < 0 || x > 1) return -std::numeric_limits<T>::infinity();
142 return std::log(pdf(x));
143 }
144
145 T cdf(T x) const override {
146 if (x <= 0) return 0;
147 if (x >= 1) return 1;
148 return boost::math::cdf(dist_, x);
149 }
150
151 T sample(boost::random::mt19937& gen) const override {
152 boost::random::beta_distribution<T> sampler(alpha_, beta_);
153 return sampler(gen);
154 }
155
156 T mean() const override { return boost::math::mean(dist_); }
157
158 T variance() const override { return boost::math::variance(dist_); }
159
160 std::optional<T> mode() const override {
161 if (alpha_ > 1 && beta_ > 1) {
162 return (alpha_ - 1) / (alpha_ + beta_ - 2);
163 } else if (alpha_ < 1 && beta_ < 1) {
164 return std::nullopt; // Bimodal
165 } else if (alpha_ == 1 && beta_ == 1) {
166 return 0.5; // Uniform, any value in [0,1]
167 }
168 return std::nullopt;
169 }
170
171 std::pair<T, T> support() const override { return {0, 1}; }
172
173 std::string name() const override {
174 return "Beta(" + std::to_string(alpha_) + ", " + std::to_string(beta_) + ")";
175 }
176
177 std::unique_ptr<PriorDistribution<T>> clone() const override {
178 return std::make_unique<BetaPrior<T>>(alpha_, beta_);
179 }
180
181 bool is_conjugate_to(const std::string& likelihood_family) const override {
182 return likelihood_family == "bernoulli" || likelihood_family == "binomial";
183 }
184
185 /**
186 * @brief Get the alpha parameter
187 */
188 T alpha() const { return alpha_; }
189
190 /**
191 * @brief Get the beta parameter
192 */
193 T beta() const { return beta_; }
194
195 private:
196 distribution_type dist_;
197 T alpha_;
198 T beta_;
199 };

201 /**
202 * @brief Normal distribution prior - conjugate for Normal likelihood with known variance
203 */
204 template <typename T = double> class NormalPrior : public PriorDistribution<T> {
205 public:
206 using distribution_type = boost::math::normal_distribution<T>;
207
208 NormalPrior(T mean, T std_dev) : dist_(mean, std_dev), mean_(mean), std_dev_(std_dev) {
209 if (std_dev <= 0) {
210 throw std::invalid_argument("Standard deviation must be positive");
211 }
212 }
213
214 T pdf(T x) const override { return boost::math::pdf(dist_, x); }
215
216 T log_pdf(T x) const override {
217 return -0.5 * std::log(2 * M_PI) - std::log(std_dev_)
218 - 0.5 * std::pow((x - mean_) / std_dev_, 2);
219 }
220
221 T cdf(T x) const override { return boost::math::cdf(dist_, x); }
222
223 T sample(boost::random::mt19937& gen) const override {
224 boost::random::normal_distribution<T> sampler(mean_, std_dev_);
225 return sampler(gen);
226 }
227
228 T mean() const override { return mean_; }
229 T variance() const override { return std_dev_ * std_dev_; }
230
231 std::optional<T> mode() const override { return mean_; }
232
233 std::pair<T, T> support() const override {
234 return {-std::numeric_limits<T>::infinity(), std::numeric_limits<T>::infinity()};
235 }
236
237 std::string name() const override {
238 return "Normal(" + std::to_string(mean_) + ", " + std::to_string(std_dev_) + ")";
239 }
240
241 std::unique_ptr<PriorDistribution<T>> clone() const override {
242 return std::make_unique<NormalPrior<T>>(mean_, std_dev_);
243 }
244
245 bool is_conjugate_to(const std::string& likelihood_family) const override {
246 return likelihood_family == "normal_known_variance";
247 }
248
249 private:
250 distribution_type dist_;
251 T mean_;
252 T std_dev_;
253 };

256 * @brief Gamma distribution prior - conjugate for Poisson and Exponential likelihoods
257 */
258 template <typename T = double> class GammaPrior : public PriorDistribution<T> {
259 public:
260 using distribution_type = boost::math::gamma_distribution<T>;
261
262 GammaPrior(T shape, T rate) : dist_(shape, 1 / rate), shape_(shape), rate_(rate) {
263 if (shape <= 0 || rate <= 0) {
264 throw std::invalid_argument("Gamma parameters must be positive");
265 }
266 }
267
268 T pdf(T x) const override {
269 if (x <= 0) return 0;
270 return boost::math::pdf(dist_, x);
271 }
272
273 T log_pdf(T x) const override {
274 if (x <= 0) return -std::numeric_limits<T>::infinity();
275 return shape_ * std::log(rate_) - std::lgamma(shape_) + (shape_ - 1) * std::log(x)
276 - rate_ * x;
277 }
278
279 T cdf(T x) const override {
280 if (x <= 0) return 0;
281 return boost::math::cdf(dist_, x);
282 }
283
284 T sample(boost::random::mt19937& gen) const override {
285 boost::random::gamma_distribution<T> sampler(shape_, 1 / rate_);
286 return sampler(gen);
287 }
288
289 T mean() const override { return shape_ / rate_; }
290 T variance() const override { return shape_ / (rate_ * rate_); }
291
292 std::optional<T> mode() const override {
293 if (shape_ >= 1) {
294 return (shape_ - 1) / rate_;
295 }
296 return std::nullopt;
297 }
298
299 std::pair<T, T> support() const override { return {0, std::numeric_limits<T>::infinity()}; }
300
301 std::string name() const override {
302 return "Gamma(" + std::to_string(shape_) + ", " + std::to_string(rate_) + ")";
303 }
304
305 std::unique_ptr<PriorDistribution<T>> clone() const override {
306 return std::make_unique<GammaPrior<T>>(shape_, rate_);
307 }
308
309 bool is_conjugate_to(const std::string& likelihood_family) const override {
310 return likelihood_family == "poisson" || likelihood_family == "exponential";
311 }
312
313 /**
314 * @brief Get the shape parameter (alpha)
315 */
316 T alpha() const { return shape_; }
317
318 /**
319 * @brief Get the rate parameter (beta)
320 */
321 T beta() const { return rate_; }
322
323 private:
324 distribution_type dist_;
325 T shape_;
326 T rate_;

329 /**
330 * @brief Uniform distribution prior - minimally informative prior
331 */
332 template <typename T = double> class UniformPrior : public PriorDistribution<T> {
333 public:
334 using distribution_type = boost::math::uniform_distribution<T>;
335
336 UniformPrior(T lower, T upper) : dist_(lower, upper), lower_(lower), upper_(upper) {
337 if (lower >= upper) {
338 throw std::invalid_argument("Lower bound must be less than upper bound");
339 }
340 }
341
342 T pdf(T x) const override {
343 if (x < lower_ || x > upper_) return 0;
344 return 1 / (upper_ - lower_);
345 }
346
347 T log_pdf(T x) const override {
348 if (x < lower_ || x > upper_) return -std::numeric_limits<T>::infinity();
349 return -std::log(upper_ - lower_);
350 }
351
352 T cdf(T x) const override {
353 if (x <= lower_) return 0;
354 if (x >= upper_) return 1;
355 return (x - lower_) / (upper_ - lower_);
356 }
357
358 T sample(boost::random::mt19937& gen) const override {
359 boost::random::uniform_real_distribution<T> sampler(lower_, upper_);
360 return sampler(gen);
361 }
362
363 T mean() const override { return (lower_ + upper_) / 2; }
364 T variance() const override {
365 T range = upper_ - lower_;
366 return (range * range) / 12;
367 }
368
369 std::optional<T> mode() const override {
370 return std::nullopt; // Any value in [lower, upper]
371 }
372
373 std::pair<T, T> support() const override { return {lower_, upper_}; }
374
375 std::string name() const override {
376 return "Uniform(" + std::to_string(lower_) + ", " + std::to_string(upper_) + ")";
377 }
378
379 std::unique_ptr<PriorDistribution<T>> clone() const override {
380 return std::make_unique<UniformPrior<T>>(lower_, upper_);
381 }
382
383 private:
384 distribution_type dist_;
385 T lower_;
386 T upper_;

389 /**
390 * @brief Factory function for creating common prior distributions
391 */
392 template <typename T = double>
393 std::unique_ptr<PriorDistribution<T>> create_prior(const std::string& distribution_name,
394 const std::vector<T>& parameters) {
395 if (distribution_name == "beta" && parameters.size() == 2) {
396 return std::make_unique<BetaPrior<T>>(parameters[0], parameters[1]);
397 } else if (distribution_name == "normal" && parameters.size() == 2) {
398 return std::make_unique<NormalPrior<T>>(parameters[0], parameters[1]);
399 } else if (distribution_name == "gamma" && parameters.size() == 2) {
400 return std::make_unique<GammaPrior<T>>(parameters[0], parameters[1]);
401 } else if (distribution_name == "uniform" && parameters.size() == 2) {
402 return std::make_unique<UniformPrior<T>>(parameters[0], parameters[1]);
403 }
404
405 throw std::invalid_argument("Unknown distribution or incorrect parameters");
406 }
